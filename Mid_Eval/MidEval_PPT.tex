% --- SLIDE 5: Literature Context ---
\overlaybox[Background: Evolution of Urban Growth Models]{
\textbf{Traditional CA Models: The SLEUTH Foundation}
\begin{itemize}
    \item \textbf{Clarke et al. (1997):} Pioneered CA-based urban growth modeling
    \item SLEUTH model components:
    \begin{itemize}
        \item \textbf{S}lope, \textbf{L}and use, \textbf{E}xclusion, \textbf{U}rban extent, \textbf{T}ransportation, \textbf{H}ill shade
        \item Fixed transition rules based on bio-physical variables
        \item Globally adopted framework for urban expansion modeling
    \end{itemize}
    \item \textbf{Strengths:}Captures spatial dynamics well
\end{itemize}
\vspace{0.5em}


}
\overlaybox[Background: Evolution of Urban Growth Models]{

\textbf{Limitations Identified (Vani \& Prasad, 2022)}
\begin{itemize}
    \item {Closed architecture:} Cannot modify or add driving factors
    \item {Limited to bio-physical drivers:} Excludes socio-economic factors like population, income
    \item {Computationally inefficient:} Calibration requires large CPU hours
    \item {Assumes uniform drivers:} Same factors everywhere, ignores local context
\end{itemize}


}
% --- SLIDE 5: Literature Context ---
\overlaybox[Background: Evolution of Urban Growth Models]{
\textbf{The Hybrid Model Solution}
\begin{itemize}
    \item \textbf{Emerging approach:} Combine CA with statistical and optimization methods
    \item \textbf{Vani \& Prasad (2022):} Developed CA-MLR-GA hybrid for Vijayawada
    \begin{itemize}
        \item MLR: Identifies statistically significant drivers
        \item CA: Captures neighborhood spatial dynamics
        \item GA: Automates parameter calibration
    \end{itemize}
    \item \textbf{Advantages:} Flexible drivers, incorporates socio-economics, context-specific
\end{itemize}
\vspace{0.3em}

\textit{Our study adapts and extends this hybrid framework for metropolitan-scale Hyderabad}
}

% --- SLIDE 8: Hybrid Framework Architecture ---
\overlaybox[Methodology Overview: The Three-Component System]{
\textbf{Why a Hybrid Approach?}
\begin{itemize}
    \item No single method captures both drivers AND spatial dynamics adequately
    \item Our solution: Integrate three complementary techniques
\end{itemize}
\vspace{0.5em}

\textbf{Component 1: Multinomial Logistic Regression (MLR)}
\begin{itemize}
    \item \textbf{Purpose:} Identify which drivers matter and how much
    \item \textbf{Input:} 5 driver variables (slope, elevation, population, roads, center)
    \item \textbf{Output:} Suitability probability for each pixel
    \item \textbf{Answers:} "Based on characteristics, how likely is this pixel to urbanize?"
\end{itemize}
\vspace{0.5em}


}
% --- SLIDE 8: Hybrid Framework Architecture ---
\overlaybox[Methodology Overview: The Three-Component System]{

\textbf{Component 2: Cellular Automata (CA)}
\begin{itemize}
    \item \textbf{Purpose:} Add neighborhood effects and spatial contagion
    \item \textbf{Input:} MLR probabilities + current urban pattern
    \item \textbf{Mechanism:} 5$\times$5 Moore kernel (distance-weighted neighbors)
    \item \textbf{Output:} Combined probability including spatial context
    \item \textbf{Answers:} "Are nearby pixels already urban? If yes, boost probability"
\end{itemize}
\vspace{0.5em}
}
% --- SLIDE 8: Hybrid Framework Architecture ---
\overlaybox[Methodology Overview: The Three-Component System]{

\textbf{Component 3: Genetic Algorithm (GA)}
\begin{itemize}
    \item \textbf{Purpose:} Automatically find optimal parameters
    \item \textbf{Optimizes:} Neighborhood weight ($\omega$), randomness ($\alpha$), iterations
    \item \textbf{Calibration:} Evolves parameters to maximize historical accuracy (1990$\rightarrow$2000)
    \item \textbf{Answers:} "What parameter values produce most accurate predictions?"
\end{itemize}
\vspace{0.5em}

}
\overlaybox[Methodology Overview: The Three-Component System]{

\textbf{Key: Quota-Based Allocation}
\begin{itemize}
    \item Traditional CA: All pixels with P $>$ 0.5 urbanize $\rightarrow$ \textcolor{Red}{over-prediction!}
    \item Our approach: Calculate demand, select TOP-K highest probability pixels
    \item Result: Correct quantity + spatial realism
\end{itemize}
}

% --- SLIDE 11: LULC Classification Methodology ---
\overlaybox[LULC Classification: Random Forest Approach]{
\textbf{Step 1: Reference Data Preparation}
\begin{itemize}
    \item Loaded ESA WorldCover 2020 (10m global land cover product)
    \item Reclassified ESA 11 classes into 4 study classes:
    \begin{itemize}
        \item Water: ESA classes 80 (water), 90 (wetland)
        \item Vegetation: ESA classes 10, 20, 30, 40, 95, 100 (trees, shrubs, grass, crops, mangroves, moss)
        \item Barren: ESA classes 60, 70 (bare, snow)
        \item Built-up: ESA class 50 (built-up)
    \end{itemize}
\end{itemize}
\vspace{0.5em}

\textbf{Step 2: Feature Extraction from Landsat 2020}
\begin{itemize}
    \item \textbf{10 spectral features:} Blue, Green, Red, NIR, SWIR1, SWIR2, NDVI, NDBI, NDWI, MNDWI
    \item Combined Landsat features with ESA labels for training
\end{itemize}
\vspace{0.5em}


}
\overlaybox[LULC Classification: Random Forest Approach]{
\textbf{Step 3: Training Sample Collection}
\begin{itemize}
    \item \textbf{Stratified sampling:} 500 points per class = 2,000 total training samples
    \item Each sample has: Landsat features (predictors) + ESA label (ground truth)
    \item Sampling seed: 42 (for reproducibility)
\end{itemize}
\vspace{0.5em}

\textbf{Step 4: Random Forest Training}
\begin{itemize}
    \item Trained on 2,000 samples from 2020 data
    \item \textbf{Parameters:} 100 trees, default variables per split
\end{itemize}
\vspace{0.5em}

\textbf{Step 5: Classification of All Years}
\begin{itemize}
    \item Applied trained RF classifier to all 7 years (1990–2020)
    \item \textbf{Output:} 7 LULC maps at 30m resolution
\end{itemize}
}

\overlaybox[Multinomial Logistic Regression: Driver Analysis]{

\textbf{MLR Mathematical Formulation (Softmax)}

For each LULC class $k$, probability is:
\[
P(Y = k | X) = \frac{\exp(\beta_{k0} + \beta_{k1}X_1 + ... + \beta_{k5}X_5)}{\sum_{j=1}^{4} \exp(\beta_{j0} + \beta_{j1}X_1 + ... + \beta_{j5}X_5)}
\]

Where:
\begin{itemize}
    \item $X_1, ..., X_5$ = driver variables (slope, elevation, population, distance to roads, distance to center)
    \item $\beta_{kp}$ = coefficient for driver $p$ in class $k$ (learned from data)
    \item Denominator ensures probabilities sum to 1 across all 4 classes
\end{itemize}

}
% --- SLIDE 21: MLR Analysis - Identifying Dominant Drivers ---
\overlaybox[Multinomial Logistic Regression: Driver Analysis]{

\textbf{Model Fit: McFadden's Pseudo R²}
\[
\text{Pseudo } R^2 = 1 - \frac{LL_{\text{model}}}{LL_{\text{null}}}
\]
\begin{itemize}
    \item $LL_{\text{model}}$ = log-likelihood of fitted model
    \item $LL_{\text{null}}$ = log-likelihood of null model (intercept only)
    \item Analogous to R² in linear regression (0 = no improvement, 1 = perfect)
\end{itemize}
\vspace{0.3em}
}

% --- SLIDE 23: Neighborhood Analysis - Moore Kernel ---
\overlaybox[Neighborhood Effect: 5×5 Moore Kernel]{
\textbf{Mathematical Formulation}

\textbf{Step 1: Calculate Euclidean distance from center}
\[
d_{ij} = \sqrt{(i - c)^2 + (j - c)^2}
\]
Where $c = 2$ (center of 5×5 grid), $i, j \in \{0, 1, 2, 3, 4\}$

\textbf{Step 2: Apply inverse distance weighting}
\[
w_{ij} = \begin{cases} 
\frac{1}{d_{ij}} & \text{if } d_{ij} > 0 \\
0 & \text{if } d_{ij} = 0 \text{ (center pixel)}
\end{cases}
\]

\textbf{Step 3: Normalize weights to sum to 1}
\[
w_{ij}^{\text{norm}} = \frac{w_{ij}}{\sum_{i,j} w_{ij}}
\]

}

    \item \textbf{7.24:} Adjacent pixels (up/down/left/right) — strongest influence
    \item \textbf{5.12:} Diagonal neighbors — medium influence
    \item \textbf{3.62, 3.24, 2.56:} Outer pixels — decreasing influence with distance
    \item \textbf{Sum = 1.0:} Normalized weights (probabilities)
\end{itemize}

\textbf{Quota-Based Allocation Process}
\begin{enumerate}
    \item Calculate total demand for target year using exponential formula
    \item Divide demand equally across iterations: $\text{quota}_{\text{iter}} = \text{Demand} / t$
    \item \textbf{Each iteration:}
    \begin{itemize}
        \item Calculate $P_{\text{MLR}}$ for all pixels (driver-based suitability)
        \item Calculate $P_{\text{neighborhood}}$ using Moore kernel (spatial contagion)
        \item Combine: $P_{\text{final}} = P_{\text{MLR}} \times (1 + \omega \times P_{\text{neighborhood}}) + \text{random}[-\alpha, \alpha]$
        \item Filter: Keep only non-urban, non-excluded, inside-circle pixels
        \item Select: TOP $\text{quota}_{\text{iter}}$ pixels by probability
        \item Convert: Selected pixels become urban (Class 4)
    \end{itemize}
\end{enumerate}

    \item \textbf{Impact:} Built-up baseline discrepancy:
    \begin{itemize}
        \item Original 2020 LULC: 471.42 km² (pre-resampling)
        \item Resampled 2020 LULC: 456.76 km² (used in predictions)
        \item Difference: -14.66 km² (-3.1\%) due to resampling artifacts
    \end{itemize}
    \item \textbf{Implication:} Minor quantitative shift, but spatial patterns preserved
\end{itemize}


}

% --- SLIDE 30: Limitations and Challenges ---
\overlaybox[Study Limitations and Methodological Challenges]{

\textbf{2. Simplified GA Optimization}
\begin{itemize}
    \item \textbf{Limitation:} Full-resolution GA calibration not completed
    \item \textbf{What we did:} Used surrogate fitness function (simplified demo)
    \begin{itemize}
        \item 20 individuals instead of 50
        \item 30 generations instead of 100
        \item Sample-based evaluation instead of full raster simulation
    \end{itemize}
    \item \textbf{Implication:} Parameters ($\omega$=1.200, $\alpha$=0.050) are preliminary, not rigorously optimized
    \item \textbf{Impact:} May not represent true optimal values; could affect prediction accuracy
\end{itemize}

}
% --- SLIDE 30: Limitations and Challenges ---
\overlaybox[Study Limitations and Methodological Challenges]{

\textbf{3. Sample-Based Calibration}
\begin{itemize}
    \item \textbf{Limitation:} Calibration used 4,126 stratified samples, not full 2.86M pixels
    \item \textbf{Impact:} 
    \begin{itemize}
        \item High reported accuracy (93.65\%) may not reflect full spatial performance
        \item Neighborhood effects calculated on samples, not actual spatial neighborhoods
        \item Full raster validation not performed
    \end{itemize}
\end{itemize}
}
% --- SLIDE 31: Additional Limitations ---
\overlaybox[Additional Limitations and Constraints]{
\textbf{4. Static Driver Assumption (2020 values frozen)}
\begin{itemize}
    \item \textbf{Limitation:} All driver variables held constant at 2020 values
    \begin{itemize}
        \item Population density: 2020 GHS-POP used for all future years
        \item Road network: 2020 OSM data (no new roads added)
        \item Slope, elevation, distance to center: naturally static
    \end{itemize}
    \item \textbf{Reality:} Infrastructure expands, population redistributes
    \item \textbf{Implication:} Underestimates growth in areas with future infrastructure development
\end{itemize}

}
% --- SLIDE 31: Additional Limitations ---
\overlaybox[Additional Limitations and Constraints]{
\textbf{5. Model Validation Constraints}
\begin{itemize}
    \item \textbf{Calibration:} 1990→2000 (historical data available)
    \item \textbf{Validation:} 2000→2010 (sample-based, not full raster)
    \item \textbf{Gap:} No validation for 2010→2020 or beyond
    \item \textbf{Impact:} Cannot assess model performance for recent decade or future
\end{itemize}
\vspace{0.3em}
}

\textbf{6. Single Scenario (Business-as-usual only)}
\begin{itemize}
    \item \textbf{Limitation:} Only BAU growth modeled at 2.67\% annual rate
\end{itemize}
}

\textbf{7. No Empirical Future Validation Possible}
\begin{itemize}
    \item \textbf{Fundamental constraint:} Predictions for 2025–2050 cannot be validated
    \item \textbf{Reliance:} Model accuracy depends on historical calibration holding into future
    \item \textbf{Assumption:} Growth patterns/drivers remain consistent (may not hold)
\end{itemize}
